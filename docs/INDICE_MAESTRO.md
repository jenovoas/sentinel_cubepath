# backend/data_analysis_suite.rs

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

class MacroDataAnalyzer:
    """
    Suite de análisis para macro cantidades de datos.

    "Necesito que analices macro cantidades de datos conmigo"
    """

    def __init__(self):
        self.datasets = {}
        self.results = {}

    def load_benchmark_data(self, filepath):
        """Carga datos de benchmarks"""
        df = pd.read_json(filepath)
        self.datasets['benchmark'] = df
        return df

    def analyze_throughput_vs_distance(self):
        """
        Analiza throughput vs distancia.

        Valida hipótesis de aceleración exponencial.
        """
        df = self.datasets['benchmark']

        # Regresión lineal
        slope, intercept, r_value, p_value, std_err = stats.linregress(
            df['distance_km'],
            df['throughput']
        )

        # ¿Es aceleración o degradación?
        if slope > 0:
            trend = "ACELERACIÓN ✅"
        else:
            trend = "DEGRADACIÓN ❌"

        return {
            'slope': slope,
            'r_squared': r_value**2,
            'p_value': p_value,
            'trend': trend
        }

    def cross_correlate_metrics(self):
        """
        Cruza correlaciones entre todas las métricas.

        Descubre patrones ocultos.
        """
        df = self.datasets['benchmark']

        # Matriz de correlación
        corr_matrix = df.corr()

        # Visualizar
        plt.figure(figsize=(12, 10))
        sns.heatmap(corr_matrix, annot=True, cmap='coolwarm')
        plt.title('Matriz de Correlación - Macro Datos')
        plt.savefig('correlation_matrix.png')

        return corr_matrix

    def detect_anomalies(self):
        """
        Detecta anomalías en datos masivos.

        Usa Isolation Forest (ML).
        """
        from sklearn.ensemble import IsolationForest

        df = self.datasets['benchmark']

        # Entrenar modelo
        clf = IsolationForest(contamination=0.1)
        predictions = clf.fit_predict(df[['throughput', 'latency']])

        # Marcar anomalías
        df['anomaly'] = predictions

        # Visualizar
        plt.figure(figsize=(10, 6))
        plt.scatter(
            df[df['anomaly'] == 1]['throughput'],
            df[df['anomaly'] == 1]['latency'],
            c='blue', label='Normal'
        )
        plt.scatter(
            df[df['anomaly'] == -1]['throughput'],
            df[df['anomaly'] == -1]['latency'],
            c='red', label='Anomalía'
        )
        plt.xlabel('Throughput')
        plt.ylabel('Latency')
        plt.legend()
        plt.title('Detección de Anomalías')
        plt.savefig('anomalies.png')

        return df[df['anomaly'] == -1]

    def generate_report(self):
        """
        Genera reporte completo de análisis.

        Para revisión humana.
        """
        report = {
            'throughput_analysis': self.analyze_throughput_vs_distance(),
            'correlations': self.cross_correlate_metrics(),
            'anomalies': self.detect_anomalies(),
            'summary': self.generate_summary()
        }

        return report
